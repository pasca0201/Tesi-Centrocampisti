u4 = np.uint32(1)

td = np.timedelta64(1, "D")
b_ = np.bool(1)

b = bool(1)
f = float(1)

AR = np.array([1], dtype=np.bool)
AR.setflags(write=False)

AR2 = np.array([1], dtype=np.timedelta64)

td % td
td % AR2

divmod(td, td)

divmod(AR2, td)
